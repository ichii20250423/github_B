Q3. 全件検索 (id, 名前, HP)

SELECT id, name, HP
FROM monster;

Q4. 経験値が1000以上

SELECT id, name, system, HP, experience
FROM monster
WHERE experience >= 1000;

Q5. ドロップアイテムがNULL

SELECT id, name, dropitem
FROM monster
WHERE dropitem IS NULL;

Q7. 系統ごとのレコード数

SELECT system, COUNT(*) AS number_of_records
FROM monster
GROUP BY system;
